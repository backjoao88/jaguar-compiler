package com.jag.parser;

terminal Integer CONSTANTE;
terminal String OP_ADICAO;
terminal String OP_SUB;

non terminal ExpSoma EXP_SOMA;
non terminal ExpSub EXP_SUB;
non terminal ExpSomaOperador EXP_SOMA_OPERADOR;
non terminal ExpSubOperador EXP_SUB_OPERADOR;


EXP_SOMA ::= EXP_SUB:exp_sub EXP_SOMA_OPERADOR:exp_soma_operador		{: RESULT = new Exp_Soma(exp_sub, exp_soma_operador); :} ;
EXP_SOMA_OPERADOR ::= OP_ADICAO EXP_SOMA:exp_soma | /* */ 	        	{: RESULT = new Exp_Soma_Operador("+", exp_soma);  :} ;


EXP_SUB ::= CONSTANTE:constante EXP_SUB_OPERADOR:exp_sub_operador		{: RESULT = new Exp_Sub(Integer, exp_sub_operador); :} ;
EXP_SUB_OPERADOR ::= OP_SUB EXP_SUB:exp_sub  | /* */        			{: RESULT = new Exp_Sub_Operador("-", exp_sub); :} ;
